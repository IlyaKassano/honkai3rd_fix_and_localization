	optimize_memory
	capital_scope = {
		create_fleet = {
			effect = {
				set_owner = root
				while = {
					count = 5
					create_army_transport = {
						ship_name = random
						#graphical_culture = "root"
						army_name = Drone
						army_type = "titan_army_s"
					}
				}
				set_location = {
					target = prev
					distance = 40
					angle = random
				}	
				set_fleet_flag = hk648_captain
			}
		}
	}
	
	capital_scope = {
		create_fleet = {
			effect = {
				set_owner = root						
				while = {
					count = 4
					create_ship = {
						name = random
						design = "Vulcan gun battleship"
					}
				}

					target = prev

				}

				set_fleet_flag = hk648_captain

				can_upgrade = yes	
			}
		}
	}
}
