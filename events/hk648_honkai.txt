# Triggers country_event for the attacker upon defeat
# This = country, attack leader
# From = country, planet owner
# FromFrom = planet
#战胜崩坏#
country_event = {
	id = hk648_honkai.4001
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		this = { is_country_type = hk_country }
	}
	
	immediate = {
		fromfrom = {
			remove_planet_flag = honkai_attack
		}
		from = {
			country_event = { 
				id = hk648_honkai.4002 
				days = 50
				random = 10
			} 
		}
		if = {	
			limit = { 
				fromfrom = { 
					OR = {
						has_planet_flag = honkai_500
						has_planet_flag = honkai_3000
						has_planet_flag = honkai_1000
					}
				}
			}
			fromfrom = {
				planet_event = {
					id = hk648_honkai.4005
				}
			}
		}
		if = {	
			limit = { 
				fromfrom = { 
					OR = {
						has_planet_flag = honkai_8000
						has_planet_flag = honkai_12000
						has_planet_flag = honkai_25000
					}
				}
			}
			fromfrom = {
				planet_event = {
					id = hk648_honkai.4006
				}
			}
		}
		if = {	
			limit = { 
				fromfrom = { 
					OR = {
						has_planet_flag = honkai_40000
						has_planet_flag = honkai_70000
						has_planet_flag = honkai_100000
					}
				}
			}
			fromfrom = {
				planet_event = {
					id = hk648_honkai.4007
				}
			}
		}
	}
}
